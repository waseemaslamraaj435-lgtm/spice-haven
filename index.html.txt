         let cartCount = 0;
         let total = 0;

         function toggleCart(){

         let cartBox = document.getElementById("cart-box");

         cartBox.style.display = cartBox.style.display === "none" ? "block" : "none";

         }

         document.querySelectorAll('.btn').forEach(button => {

         button.addEventListener('click', function(){

         const productName = this.parentElement.querySelector('.product-name').textContent;

         const productPrice = parseFloat(
         this.parentElement.querySelector('.product-price').textContent.replace('$','')
         );

         cartCount++;

         total += productPrice;

         document.getElementById("cart-count").textContent = cartCount;
 
         document.getElementById("cart-total").textContent = total.toFixed(2);

         let li = document.createElement("li");

         li.innerHTML = productName + " $" + productPrice + 
      ' <button onclick="removeItem(this,'+productPrice+')">❌</button>';

        document.getElementById("cart-items").appendChild(li);

        });

        });

        function removeItem(button,price){

        button.parentElement.remove();

        cartCount--;

        total -= price;

        document.getElementById("cart-count").textContent = cartCount;

        document.getElementById("cart-total").textContent = total.toFixed(2);

        }
